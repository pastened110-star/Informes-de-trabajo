import streamlit as st
from fpdf import FPDF
from datetime import datetime

st.set_page_config(page_title="ReportePro - Informes Técnicos", page_icon="📝")

st.title("📝 Generador de Informes Técnicos")
st.markdown("Herramienta gratuita para generar reportes PDF profesionales en terreno.")

# DATOS DEL EMISOR
with st.expander("🏗️ Datos de la Empresa / Técnico", expanded=True):
    col_em1, col_em2 = st.columns(2)
    with col_em1:
        nombre_empresa = st.text_input("Nombre o Razón Social", placeholder="Ej: Servicios Eléctricos SPA")
        rut_empresa = st.text_input("RUT", placeholder="12.345.678-9")
    with col_em2:
        contacto_em = st.text_input("Teléfono / Email", placeholder="contacto@empresa.cl")

# DETALLES DEL INFORME
st.subheader("📋 Detalles del Servicio")
col1, col2 = st.columns(2)
with col1:
    proyecto = st.text_input("Nombre del Proyecto / Cliente")
    fecha = st.date_input("Fecha", value=datetime.now())
with col2:
    ubicacion = st.text_input("Ubicación / Faena")
    tipo_servicio = st.selectbox("Tipo de Servicio", 
                                ["Inspección Normativa RIC", "Mantenimiento Preventivo", 
                                 "Mantenimiento Correctivo", "Levantamiento Técnico", "Otro"])

hallazgos = st.text_area("Descripción de los trabajos / Hallazgos técnicos")
conclusiones = st.text_area("Conclusiones y Recomendaciones")

# CLASE PDF
class PDF(FPDF):
    def header(self):
        self.set_font('Arial', 'B', 12)
        self.cell(0, 10, 'INFORME TECNICO DE TERRENO', 0, 1, 'C')
        self.ln(5)

def generar_pdf(empresa, rut, contacto, proyecto, fecha, ubicacion, tipo, obs, concl):
    pdf = PDF()
    pdf.add_page()
    pdf.set_font("Arial", 'B', 10)
    pdf.cell(0, 8, f"EMISOR: {empresa.upper()} - RUT: {rut}", ln=True)
    pdf.set_font("Arial", '', 9)
    pdf.cell(0, 5, f"Contacto: {contacto}", ln=True)
    pdf.line(10, 35, 200, 35)
    pdf.ln(10)

    pdf.set_font("Arial", 'B', 11)
    pdf.cell(0, 10, "DATOS DEL PROYECTO", ln=True)
    pdf.set_font("Arial", '', 10)
    pdf.cell(0, 7, f"Proyecto/Cliente: {proyecto}", ln=True)
    pdf.cell(0, 7, f"Ubicacion: {ubicacion}", ln=True)
    pdf.cell(0, 7, f"Fecha: {fecha} | Tipo de Servicio: {tipo}", ln=True)
    pdf.ln(5)

    pdf.set_font("Arial", 'B', 11)
    pdf.cell(0, 10, "DESCRIPCION TECNICA", ln=True)
    pdf.set_font("Arial", '', 10)
    pdf.multi_cell(0, 7, obs)
    pdf.ln(5)

    pdf.set_font("Arial", 'B', 11)
    pdf.cell(0, 10, "RECOMENDACIONES", ln=True)
    pdf.set_font("Arial", '', 10)
    pdf.multi_cell(0, 7, concl)

    # El 'latin-1' ayuda con los acentos básicos en FPDF
    return pdf.output(dest='S').encode('latin-1', 'ignore')

if st.button("🚀 Generar Informe Profesional"):
    if nombre_empresa and proyecto and hallazgos:
        pdf_bytes = generar_pdf(nombre_empresa, rut_empresa, contacto_em, proyecto, fecha, ubicacion, tipo_servicio, hallazgos, conclusiones)
        st.download_button(label="⬇️ Descargar PDF", data=pdf_bytes, file_name=f"Informe_{proyecto}.pdf", mime="application/pdf")
    else:
        st.warning("Completa los campos obligatorios.")
